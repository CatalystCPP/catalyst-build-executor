80[[clang::always_inline]]
81bool inline Executor::needs_rebuild(const BuildStep &step, StatCache &stat_cache) {
82 if (std::filesystem::exists(step.output)) {
83 auto output_modtime = std::filesystem::last_write_time(step.output);
84
85 if (stat_cache.changed_since(config.build_file, output_modtime)) {
86 return true;
87 }
88
89 if (step.depfile_inputs.has_value()) {
90 for (const auto &dep : *step.depfile_inputs) {
91 if (stat_cache.changed_since(std::filesystem::path(dep), output_modtime)) {
92 return true;
93 }
94 }
95 }
96 if (step.opaque_inputs.has_value()) {
97 for (const auto &opaque : *step.opaque_inputs) {
98 if (stat_cache.changed_since(std::filesystem::path(opaque), output_modtime)) {
99 return true;
100 }
101 }
102 }
103 // this is our way of making sure that the .d file isn't stale
104 for (const auto &input : step.parsed_inputs) {
105 if (stat_cache.changed_since(input, output_modtime)) {
106 return true;
107 }
108 }
109 return false;
110 }
111 return true;
112}

440 auto worker = [&]() {
441 while (true) {
442 size_t node_idx;
443 {
444 std::unique_lock lock(mtx);
445 cv_ready.wait(lock, [&] {
446 return !ready_queue.empty() || completed_count == total_nodes || active_workers == 0; //
447 });
448
449 if (ready_queue.empty()) {
450 if (completed_count == total_nodes)
451 return;
452 // If queue is empty and no active workers, but work not done -> Cycle
453 return;
454 }
455
456 node_idx = ready_queue.top().node_idx;
457 ready_queue.pop();
458 active_workers++;
459 }
460
461 int result = process_step(node_idx);
462
463 {
464 std::lock_guard lock(mtx);
465 active_workers--;
466
467 size_t new_work_count = 0;
468
469 if (result != 0) {
470 error_occurred = true;
471 completed_count = total_nodes; // Force exit
472 } else {
473 completed_count.fetch_add(1, std::memory_order_relaxed);
474 const auto &node = build_graph.nodes()[node_idx];
475 for (size_t neighbor : node.out_edges) {
476 in_degrees[neighbor]--;
477 if (in_degrees[neighbor] == 0) {
478 push_ready(neighbor);
479 new_work_count++;
480 }
481 }
482 }
483
484 bool build_finished = (completed_count == total_nodes);
485 bool stall_detected = (active_workers == 0);
486 constexpr auto TUNABLE__notify_all_criteria = 10;
487 if (build_finished || error_occurred || stall_detected) {
488 cv_ready.notify_all();
489 } else if (new_work_count == 1) {
490 cv_ready.notify_one();
491 } else if (new_work_count >= TUNABLE__notify_all_criteria) {
492 cv_ready.notify_all();
493 } else {
494 for (size_t ii = 0; ii < new_work_count; ++ii) {
495 cv_ready.notify_one();
496 }
497 }
498 }
499 }
500 };
501
502 size_t thread_count = config.jobs;
503 if (thread_count == 0)
504 thread_count = std::thread::hardware_concurrency();
505 if (thread_count == 0)
506 thread_count = 1;
507
508 for (size_t i = 0; i < thread_count; ++i) {
509 pool.emplace_back(worker);
510 }
511
512 pool.clear(); // Join all threads
513
514 if (error_occurred)
515 return std::unexpected("Build Failed");
516
517 if (completed_count != total_nodes)
518 return std::unexpected("Cycle detected: Build stalled with pending nodes.");
519
520 return {};
521}
